```C
int cmp(const void* _a, const void* _b) {
    int a = *(int*)_a, b = *(int*)_b;
    return a - b;
}

bool containsDuplicate(int* nums, int numsSize) {
    qsort(nums, numsSize, sizeof(int), cmp);
    for (int i = 0; i < numsSize - 1; i++) {
        if (nums[i] == nums[i + 1]) {
            return true;
        }
    }
    return false;
}
```
>用了快排函数，然后经过一次遍历确定相邻的数不相等即可

```C
//接下来来分析一下快排函数的手动实现
#include <stdio.h>

int a[100], n, temp;

void QuickSort(int h, int t)
{
	//h为首位置，t为末尾位置
    if(h >= t) return; //确保h < n
     
     int mid = (h + t) / 2, i = h, j = t, x;
     x = a[mid];
     while(1)
     {
         while(a[i] < x)
             i++;
         while(a[j] > x) 
             j--;
         if(i >= j) 
             break;
         temp = a[i];
         a[i] = a[j];
         a[j] = temp;
     }
     a[j] = x;
     QuickSort(h, j - 1);
     QuickSort(j + 1, t);
     return ;
}

int main()
{
     int i;
     scanf("%d", &n);
     for(i = 0; i < n; i++)
         scanf("%d", &a[i]);
     QuickSort(0, n - 1);
     for(i = 0; i < n; i++) 
         printf("%d ", a[i]);
     return 0;
}
```
